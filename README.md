# wc_command using System Calls

argv[1] = sample.txt\
argv[2] = "-l"\
argv[3] (optional) = word to count occurrence of (only used in case of -wrd)\

argv[2] possible values: \
characters (-c)\
lines (-l)\
words (-w)\
special characters (-sc)\
comment lines (-cl)\
paragraphs (-p)\
bytes (-b)\
occurrences of particular word (-wrd)\
longest line (-ll)
